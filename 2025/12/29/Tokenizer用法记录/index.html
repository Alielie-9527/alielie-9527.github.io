Tokenizer用法记录

2025/12/29 LLM学习 Tokenizer 大模型基础

Tokenizer组件用法记录

导入分词器

from transformers import AutoTokenizer

tokenizer = AutoTokenizer.from_pretrained('bert-base-uncased')

常用接口

  • tokenizer.get_vocab()：获取词汇表
# 词汇表示例
vocab = {
    '[PAD]': 0,      # 填充标记
    '[UNK]': 1,      # 未知词标记
    '[CLS]': 2,      # 分类标记
    '[SEP]': 3,      # 分隔标记
    '[MASK]': 4,     # 掩码标记
    'hello': 7592,   # 普通词
    'world': 2088,   # 普通词
    # ... 更多词汇
}

  • tokenizer.tokenize: 进行分词

  • tokenizer.encode:进行编码

  • tokenizer.decode:进行解码

text = "Hello, how are you?"
# 分词
tokens = tokenizer.tokenize(text)  # 示例：['hello', ',', 'how', 'are', 'you', '?']
encoded = tokenizer.encode(text)   # [101, 7592, 1010, 2129, 2024, 2017, 1026, 102]

# 编码
result = tokenizer(text, return_tensors='pt', padding=True, truncation=True)
# result: {'input_ids': tensor, 'attention_mask': tensor}

# 解码
decoded = tokenizer.decode(result['input_ids'][0])
print(decoded)  # '[CLS] hello, how are you? [SEP]'

预定义特殊标记：

# 可以直接访问的属性，数字不同分词器可能不同
print(f"pad_token: {tokenizer.pad_token}")        # '[PAD]'
print(f"pad_token_id: {tokenizer.pad_token_id}")  # 0 
print(f"cls_token: {tokenizer.cls_token}")        # '[CLS]'
print(f"cls_token_id: {tokenizer.cls_token_id}")  # 101
print(f"sep_token: {tokenizer.sep_token}")        # '[SEP]'
print(f"sep_token_id: {tokenizer.sep_token_id}")  # 102
print(f"mask_token: {tokenizer.mask_token}")      # '[MASK]'
print(f"mask_token_id: {tokenizer.mask_token_id}")# 103
  • tokenizer.convert_tokens_to_ids: token转id
  • tokenizer.convert_ids_to_tokens: id 转token
# 一般token需要转换
hello_id = tokenizer.convert_tokens_to_ids('hello')  # 7592
hello_token = tokenizer.convert_ids_to_tokens(7592)  # 'hello'
  • tokenizer.add_special_tokens() ：添加特殊token
## 方法的定义
def add_special_tokens(
    self,
    special_tokens_dict: dict[str, Union[str, AddedToken, Sequence[Union[str, AddedToken]]]],  # 特殊token字典
    replace_extra_special_tokens=True,  # 是否替换额外特殊token
) -> int:  # 返回添加的token数量

# 使用
 tokenizer.add_special_tokens({
    'additional_special_tokens': ['<PROMPT_0>', '<PROMPT_1>']
})
